1. Get the Alertmanager URL:

  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name=alertmanager,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:9093 to access your Alertmanager"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9093:9093
